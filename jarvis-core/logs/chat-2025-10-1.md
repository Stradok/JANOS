from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import yaml
import modules as modules_pkg

app = FastAPI(title="Jarvis Core Brain - v0.1")

def load_config(path="config.yaml"):
    with open(path, "r") as f:
        return yaml.safe_load(f)

config = load_config()

# Build enabled modules from config and registry
ENABLED_MODULES = {}
for name, enabled in config.get("features", {}).items():
    if enabled and name in modules_pkg.MODULES:
        ENABLED_MODULES[name] = modules_pkg.MODULES[name]

class ProcessRequest(BaseModel):
    module: str
    input: dict = {}

@app.get("/modules")
def list_modules():
    return {
        "enabled": list(ENABLED_MODULES.keys()),
        "available": list(modules_pkg.MODULES.keys())
    }

@app.post("/process")
def process(req: ProcessRequest):
    module_name = req.module
    if module_name not in ENABLED_MODULES:
        raise HTTPException(status_code=404, detail=f"Module '{module_name}' is not enabled or not found.")
    module = ENABLED_MODULES[module_name]
    try:
        output = module.process(req.input)
        return {"module": module_name, "output": output}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

class MathModule(ModuleBase):
    def __init__(self):
        super().__init__("math")

    def process(self, input_data):
        # Expecting {"a": number, "b": number, "op": "add/sub/mul/div"}
        try:
            a = input_data.get("a")
            b = input_data.get("b")
            op = input_data.get("op", "add")

            if op == "add":
                result = a + b
            elif op == "sub":
                result = a - b
            elif op == "mul":
                result = a * b
            elif op == "div":
                if b == 0:
                    return {"error": "Division by zero"}
                result = a / b
            else:
                return {"error": f"Unknown operation: {op}"}

            return {
                "status": "ok",
                "operation": op,
                "a": a,
                "b": b,
                "result": result
            }
        except Exception as e:
            return {"status": "error", "message": str(e)}

        """Convert city name → latitude & longitude using Open-Meteo Geocoding API"""

            data = response.json()
            if "results" in data and len(data["results"]) > 0:
                result = data["results"][0]
                return result["latitude"], result["longitude"]

            return {"error": "Please provide either 'city' or 'lat' & 'lon'."}

            data = response.json()

            if "current_weather" not in data:
                return {"error": "Could not fetch weather."}

            weather = data["current_weather"]

import json
import os
from .base import ModuleBase

class NotesModule(ModuleBase):
    def __init__(self):
        super().__init__("notes")
        self.file_path = "memory/notes.json"
        os.makedirs("memory", exist_ok=True)
        if not os.path.exists(self.file_path):
            with open(self.file_path, "w") as f:
                json.dump([], f)

    def _read_notes(self):
        with open(self.file_path, "r") as f:
            return json.load(f)

    def _write_notes(self, notes):
        with open(self.file_path, "w") as f:
            json.dump(notes, f, indent=2)

    def process(self, input_data):
        action = input_data.get("action", "list")

        if action == "add":
            text = input_data.get("text")
            if not text:
                return {"error": "Missing note text"}
            notes = self._read_notes()
            notes.append(text)
            self._write_notes(notes)
            return {"status": "ok", "message": f"Note added: {text}"}

        elif action == "list":
            notes = self._read_notes()
            return {"status": "ok", "notes": notes}

        elif action == "delete":
            index = input_data.get("index")
            notes = self._read_notes()
            if index is not None and 0 <= index < len(notes):
                removed = notes.pop(index)
                self._write_notes(notes)
                return {"status": "ok", "message": f"Deleted note: {removed}"}
            return {"error": "Invalid note index"}

        else:
            return {"error": f"Unknown action: {action}"}

import json
import os
from .base import ModuleBase

class NotesModule(ModuleBase):
    def __init__(self):
        super().__init__("notes")
        self.file_path = "memory/notes.json"
        os.makedirs("memory", exist_ok=True)
        if not os.path.exists(self.file_path):
            with open(self.file_path, "w") as f:
                json.dump([], f)

    def _read_notes(self):
        with open(self.file_path, "r") as f:
            return json.load(f)

    def _write_notes(self, notes):
        with open(self.file_path, "w") as f:
            json.dump(notes, f, indent=2)

    def process(self, input_data):
        action = input_data.get("action", "list")

        if action == "add":
            text = input_data.get("text")
            speaker = input_data.get("speaker", "Unknown")  # 👈 default speaker
            if not text:
                return {"error": "Missing note text"}

            notes = self._read_notes()
            notes.append({"speaker": speaker, "note": text})
            self._write_notes(notes)
            return {"status": "ok", "message": f"Note added for {speaker}: {text}"}

        elif action == "list":
            notes = self._read_notes()
            return {"status": "ok", "notes": notes}

        elif action == "delete":
            index = input_data.get("index")
            notes = self._read_notes()
            if index is not None and 0 <= index < len(notes):
                removed = notes.pop(index)
                self._write_notes(notes)
                return {"status": "ok", "message": f"Deleted note: {removed}"}
            return {"error": "Invalid note index"}

        else:
            return {"error": f"Unknown action: {action}"}

import json
import os
from datetime import datetime
from .base import ModuleBase

class TasksModule(ModuleBase):
    def __init__(self):
        super().__init__("tasks")
        self.file_path = "memory/tasks.json"
        os.makedirs("memory", exist_ok=True)
        if not os.path.exists(self.file_path):
            with open(self.file_path, "w") as f:
                json.dump([], f)

    def _read_tasks(self):
        with open(self.file_path, "r") as f:
            return json.load(f)

    def _write_tasks(self, tasks):
        with open(self.file_path, "w") as f:
            json.dump(tasks, f, indent=2)

    def process(self, input_data):
        action = input_data.get("action", "list")

        if action == "add":
            text = input_data.get("text")
            deadline = input_data.get("deadline")  # e.g. "2025-10-03"
            if not text:
                return {"error": "Missing task text"}
            tasks = self._read_tasks()
            task = {
                "task": text,
                "status": "pending",
                "deadline": deadline if deadline else None,
                "created_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            }
            tasks.append(task)
            self._write_tasks(tasks)
            return {"status": "ok", "message": f"Task added: {text}"}

        elif action == "list":
            tasks = self._read_tasks()
            return {"status": "ok", "tasks": tasks}

        elif action == "complete":
            index = input_data.get("index")
            tasks = self._read_tasks()
            if index is not None and 0 <= index < len(tasks):
                tasks[index]["status"] = "completed"
                self._write_tasks(tasks)
                return {"status": "ok", "message": f"Task completed: {tasks[index]['task']}"}
            return {"error": "Invalid task index"}

        elif action == "delete":
            index = input_data.get("index")
            tasks = self._read_tasks()
            if index is not None and 0 <= index < len(tasks):
                removed = tasks.pop(index)
                self._write_tasks(tasks)
                return {"status": "ok", "message": f"Deleted task: {removed['task']}"}
            return {"error": "Invalid task index"}

        else:
            return {"error": f"Unknown action: {action}"}
